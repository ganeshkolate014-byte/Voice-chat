// Firestore Security Rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow anyone signed in to read room list
    match /rooms/{roomId} {
      allow read: if request.auth != null;
      // Allow creation if signed in
      allow create: if request.auth != null;
      // Optional: Only allow creator to update/delete (if you add that logic later)
      allow update, delete: if request.auth != null && request.auth.uid == resource.data.createdByUid;
    }
  }
}

// Realtime Database Security Rules
{
  "rules": {
    "rooms": {
      "$roomId": {
        "participants": {
          ".read": "auth != null",
          "$peerId": {
            // Users can only write their own entry
            ".write": "auth != null",
            // Optional validation: ".validate": "newData.hasChildren(['displayName', 'photoURL'])"
          }
        }
      }
    },
    "chats": {
      "$roomId": {
        ".read": "auth != null",
        ".write": "auth != null",
        // Optional: Validate message structure
        "$messageId": {
          ".validate": "newData.hasChildren(['text', 'sender', 'timestamp'])"
        }
      }
    }
  }
}
